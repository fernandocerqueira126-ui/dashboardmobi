  <name>Task 2: Criar página Proposta.tsx com busca RAG e gerador Gemini</name>
  <files>src/pages/Proposta.tsx</files>
  <action>
    Criar `src/pages/Proposta.tsx` com a seguinte estrutura completa:

    ```typescript
    import { useState, useEffect } from "react";
    import { useParams, useNavigate } from "react-router-dom";
    import { Header } from "@/components/layout/Header";
    import { Button } from "@/components/ui/button";
    import { Input } from "@/components/ui/input";
    import { Textarea } from "@/components/ui/textarea";
    import { Badge } from "@/components/ui/badge";
    import { Separator } from "@/components/ui/separator";
    import { supabase } from "@/integrations/supabase/client";
    import {
      FileText, Search, Sparkles, ArrowLeft,
      Home, DollarSign, Loader2, User, Phone
    } from "lucide-react";
    import { toast } from "sonner";

    interface LeadData {
      id: string;
      name: string;
      phone: string | null;
      email: string | null;
      value: number | null;
      link_imovel_interesse: string | null;
    }

    interface ImovelResult {
      id: string;
      name: string;
      link_imovel_interesse: string | null;
      value: number | null;
    }

    export default function Proposta() {
      const { id } = useParams<{ id: string }>();
      const navigate = useNavigate();

      const [lead, setLead] = useState<LeadData | null>(null);
      const [isLoadingLead, setIsLoadingLead] = useState(true);

      // RAG search state
      const [searchQuery, setSearchQuery] = useState("");
      const [searchResults, setSearchResults] = useState<ImovelResult[]>([]);
      const [isSearching, setIsSearching] = useState(false);
      const [selectedImovel, setSelectedImovel] = useState<ImovelResult | null>(null);

      // Gemini state
      const [instructions, setInstructions] = useState("");
      const [minuta, setMinuta] = useState("");
      const [isGenerating, setIsGenerating] = useState(false);

      // Load lead data
      useEffect(() => {
        if (!id) return;
        (async () => {
          setIsLoadingLead(true);
          const { data, error } = await supabase
            .from('leads')
            .select('id, name, phone, email, value, link_imovel_interesse')
            .eq('id', id)
            .single();
          if (error) {
            toast.error("Lead não encontrado.");
            navigate('/leads');
            return;
          }
          setLead(data as LeadData);
          setIsLoadingLead(false);
        })();
      }, [id, navigate]);

      // RAG search — debounced on input change
      useEffect(() => {
        if (searchQuery.length < 2) {
          setSearchResults([]);
          return;
        }
        const timer = setTimeout(async () => {
          setIsSearching(true);
          const { data } = await supabase
            .from('leads')
            .select('id, name, link_imovel_interesse, value')
            .ilike('link_imovel_interesse', `%${searchQuery}%`)
            .limit(8);
          setSearchResults((data as ImovelResult[]) || []);
          setIsSearching(false);
        }, 350);
        return () => clearTimeout(timer);
      }, [searchQuery]);

      const handleGenerateMinuta = async () => {
        if (!lead) return;
        const apiKey = import.meta.env.VITE_GEMINI_API_KEY;
        if (!apiKey) {
          toast.error("Configure VITE_GEMINI_API_KEY no arquivo .env");
          return;
        }

        setIsGenerating(true);
        setMinuta("");

        const imovelInfo = selectedImovel
          ? `Imóvel de interesse: Referência ${selectedImovel.link_imovel_interesse} — Valor: R$ ${Number(selectedImovel.value).toLocaleString('pt-BR')}`
          : lead.link_imovel_interesse
          ? `Imóvel de interesse: ${lead.link_imovel_interesse}`
          : "Imóvel a definir";

        const prompt = `Você é um corretor imobiliário especialista. Redija uma minuta de proposta comercial formal e persuasiva em português brasileiro.

Dados do cliente:
- Nome: ${lead.name}
- Telefone: ${lead.phone || 'não informado'}
- ${imovelInfo}
- Valor de interesse: R$ ${Number(lead.value || 0).toLocaleString('pt-BR')}

Instruções adicionais do corretor: ${instructions || 'Nenhuma instrução adicional.'}

Escreva a proposta completa com saudação, apresentação do imóvel, condições comerciais sugeridas, prazo de validade da proposta (5 dias úteis) e encerramento formal. Use linguagem profissional e calorosa.`;

        try {
          const response = await fetch(
            `https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${apiKey}`,
            {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({
                contents: [{ parts: [{ text: prompt }] }]
              })
            }
          );
          if (!response.ok) throw new Error(`Gemini API error: ${response.status}`);
          const result = await response.json();
          const text = result.candidates?.[0]?.content?.parts?.[0]?.text ?? '';
          setMinuta(text);
          toast.success("Minuta gerada com sucesso!");
        } catch (error) {
          console.error("Gemini error:", error);
          toast.error("Erro ao gerar minuta. Verifique a chave da API.");
        } finally {
          setIsGenerating(false);
        }
      };

      if (isLoadingLead) {
        return (
          <div className="flex-1 flex items-center justify-center h-screen">
            <Loader2 className="w-8 h-8 animate-spin text-primary" />
          </div>
        );
      }

      return (
        <div className="flex flex-col h-screen">
          <Header
            title="Criar Proposta"
            subtitle={`Para o lead: ${lead?.name}`}
            icon={<FileText className="w-5 h-5" />}
          />

          <div className="flex-1 overflow-auto p-6 max-w-3xl mx-auto w-full space-y-6">
            {/* Back button */}
            <Button variant="ghost" size="sm" className="gap-1.5 -ml-2" onClick={() => navigate('/leads')}>
              <ArrowLeft className="w-4 h-4" />
              Voltar ao CRM
            </Button>

            {/* Lead info card */}
            <div className="bg-card border border-border rounded-xl p-4 space-y-2">
              <h2 className="text-xs font-semibold text-muted-foreground uppercase tracking-wider">Lead</h2>
              <div className="flex items-center gap-3">
                <div className="w-10 h-10 rounded-full bg-primary/10 flex items-center justify-center text-primary font-bold text-sm">
                  {lead?.name?.[0]?.toUpperCase()}
                </div>
                <div>
                  <p className="font-semibold text-foreground flex items-center gap-1.5">
                    <User className="w-3.5 h-3.5 text-muted-foreground" />
                    {lead?.name}
                  </p>
                  {lead?.phone && (
                    <p className="text-sm text-muted-foreground flex items-center gap-1.5">
                      <Phone className="w-3 h-3" />
                      {lead.phone}
                    </p>
                  )}
                </div>
                {lead?.value && (
                  <Badge variant="outline" className="ml-auto text-warning border-warning/30">
                    <DollarSign className="w-3 h-3 mr-1" />
                    {Number(lead.value).toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' })}
                  </Badge>
                )}
              </div>
            </div>

            <Separator />

            {/* RAG Imovel Search */}
            <div className="space-y-3">
              <h2 className="text-xs font-semibold text-muted-foreground uppercase tracking-wider flex items-center gap-2">
                <Home className="w-3.5 h-3.5" />
                Buscar Imóvel (RAG)
              </h2>
              <div className="relative">
                <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />
                <Input
                  placeholder="Digite a referência do imóvel (ex: CA522)..."
                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}
                  className="pl-9 bg-secondary border-border"
                />
                {isSearching && (
                  <Loader2 className="absolute right-3 top-1/2 -translate-y-1/2 w-4 h-4 animate-spin text-muted-foreground" />
                )}
              </div>

              {searchResults.length > 0 && (
                <div className="bg-secondary/50 border border-border rounded-lg overflow-hidden">
                  {searchResults.map((result) => (
                    <button
                      key={result.id}
                      onClick={() => {
                        setSelectedImovel(result);
                        setSearchQuery(result.link_imovel_interesse || "");
                        setSearchResults([]);
                      }}
                      className={`w-full flex items-center justify-between px-4 py-3 hover:bg-primary/5 transition-colors border-b border-border/50 last:border-0 text-left ${
                        selectedImovel?.id === result.id ? 'bg-primary/10' : ''
                      }`}
                    >
                      <div>
                        <p className="text-sm font-medium text-foreground">{result.name}</p>
                        <p className="text-xs text-primary/80 font-medium">Ref: {result.link_imovel_interesse || 'N/A'}</p>
                      </div>
                      {result.value && (
                        <span className="text-xs font-semibold text-warning">
                          {Number(result.value).toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' })}
                        </span>
                      )}
                    </button>
                  ))}
                </div>
              )}

              {selectedImovel && (
                <div className="flex items-center gap-2 p-3 rounded-lg bg-primary/10 border border-primary/20">
                  <Home className="w-4 h-4 text-primary" />
                  <span className="text-sm text-primary font-medium">
                    Selecionado: Ref. {selectedImovel.link_imovel_interesse} — {selectedImovel.name}
                  </span>
                  <Button variant="ghost" size="sm" className="ml-auto h-6 text-xs" onClick={() => { setSelectedImovel(null); setSearchQuery(""); }}>
                    Limpar
                  </Button>
                </div>
              )}
            </div>

            <Separator />

            {/* AI Instructions */}
            <div className="space-y-3">
              <h2 className="text-xs font-semibold text-muted-foreground uppercase tracking-wider flex items-center gap-2">
                <Sparkles className="w-3.5 h-3.5" />
                Instruções para a IA
              </h2>
              <Textarea
                placeholder="Ex: Enfatize a localização privilegiada. Cliente prefere pagamento parcelado. Mencione que há possibilidade de desconto para assinatura rápida..."
                value={instructions}
                onChange={(e) => setInstructions(e.target.value)}
                className="bg-secondary border-border min-h-[100px] resize-none"
                rows={4}
              />
              <Button
                onClick={handleGenerateMinuta}
                disabled={isGenerating}
                className="w-full btn-primary gap-2"
                size="sm"
              >
                {isGenerating ? (
                  <><Loader2 className="w-4 h-4 animate-spin" />Gerando minuta...</>
                ) : (
                  <><Sparkles className="w-4 h-4" />Gerar Minuta</>
                )}
              </Button>
            </div>

            {/* Generated Minuta */}
            {minuta && (
              <>
                <Separator />
                <div className="space-y-3">
                  <h2 className="text-xs font-semibold text-muted-foreground uppercase tracking-wider flex items-center gap-2">
                    <FileText className="w-3.5 h-3.5" />
                    Minuta Gerada
                  </h2>
                  <div className="bg-secondary/50 border border-border rounded-xl p-5">
                    <Textarea
                      value={minuta}
                      onChange={(e) => setMinuta(e.target.value)}
                      className="bg-transparent border-0 p-0 focus-visible:ring-0 text-sm leading-relaxed min-h-[300px] resize-none"
                    />
                  </div>
                  <div className="flex gap-2">
                    <Button
                      variant="outline"
                      size="sm"
                      className="gap-1.5"
                      onClick={() => { navigator.clipboard.writeText(minuta); toast.success("Copiado!"); }}
                    >
                      Copiar texto
                    </Button>
                    <Button
                      variant="ghost"
                      size="sm"
                      className="gap-1.5"
                      onClick={handleGenerateMinuta}
                      disabled={isGenerating}
                    >
                      Regenerar
                    </Button>
                  </div>
                </div>
              </>
            )}
          </div>
        </div>
      );
    }
    ```
